## 修复执行脚本

等文件释放后，运行以下脚本一键修复：

```python
import zipfile
import shutil

docx_path = '论文_格式修正版.docx'
backup_path = '论文_格式修正版_backup_final.docx'
shutil.copy(docx_path, backup_path)

with zipfile.ZipFile(docx_path, 'r') as z:
    content = z.read('word/document.xml').decode('utf-8')

# P0: 英文摘要数据
replacements = [
    ('11.0 percentage points (70.1%)', '17.5 percentage points (70.3%)'),
    ('3.8 percentage points (24.2%)', '9.8 percentage points (39.4%)'),
    ('0.9 percentage points (5.7%)', '-2.4 percentage points (-9.6%)'),
]

for old, new in replacements:
    content = content.replace(old, new)

# P1: 表格引用格式
for i in range(5, 11):
    content = content.replace(f'表5-{i}', f'表 5-{i}')

# P1: 空格问题
content = content.replace('6总结与展望', '6 总结与展望')

# 保存
with zipfile.ZipFile(docx_path, 'w', zipfile.ZIP_DEFLATED) as zf:
    old_zip = zipfile.ZipFile(backup_path, 'r')
    for item in old_zip.infolist():
        data = old_zip.read(item.filename)
        if item.filename == 'word/document.xml':
            zf.writestr(item, content.encode('utf-8'))
        else:
            zf.writestr(item, data)
    old_zip.close()

print('修复完成！')
```
